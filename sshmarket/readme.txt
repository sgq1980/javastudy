2.proxool-cglib 安装本地库

<dependency>
	<groupId>org.logicalcobwebs.proxool</groupId>
	<artifactId>proxool-cglib</artifactId>
	<version>0.9.1</version>
	<scope>compile</scope>
</dependency>
